%% Define two different macros: A Scene Change (Leaves) and an Alternate Change (Daggers)

\newcommand{\scenechange}{\renewcommand{\pfbreakdisplay}{%
				\vspace{10pt}
				\textleaf\quad\textleaf\quad\textleaf
				\vspace{10pt}
				}
				\fancybreak*{\pfbreakdisplay}}

\newcommand{\altchange}{\renewcommand{\pfbreakdisplay}{%
				\vspace{10pt}
				\textdagger\quad\textdagger\quad\textdagger
				\vspace{10pt}
                                }
				\fancybreak*{\pfbreakdisplay}}
